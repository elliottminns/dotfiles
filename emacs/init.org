#+title: Elliott's Emacs Configuration

This is the actual Emacs configuration. =init.el= is intentionally small: it
tangles this file into an Emacs Lisp file under the XDG cache directory, then
loads that generated file.

* Paths

Keep generated state out of the symlinked dotfiles directory. Cache files go
under =$XDG_CACHE_HOME/emacs= and durable editor state goes under
=$XDG_STATE_HOME/emacs=.

#+begin_src emacs-lisp
  (defconst dreams/cache-dir
    (expand-file-name "emacs/" (or (getenv "XDG_CACHE_HOME") "~/.cache/")))

  (defconst dreams/state-dir
    (expand-file-name "emacs/" (or (getenv "XDG_STATE_HOME") "~/.local/state/")))

  (dolist (dir (list dreams/cache-dir
                     dreams/state-dir
                     (expand-file-name "auto-save/" dreams/cache-dir)
                     (expand-file-name "auto-save-list/" dreams/cache-dir)
                     (expand-file-name "backups/" dreams/cache-dir)
                     (expand-file-name "undo-tree/" dreams/state-dir)))
    (unless (file-directory-p dir)
      (make-directory dir t)))

  (setq custom-file (expand-file-name "custom.el" dreams/state-dir))
  (load custom-file t)

  (setq auto-save-list-file-prefix (expand-file-name "auto-save-list/.saves-" dreams/cache-dir))
  (setq backup-directory-alist `(("." . ,(expand-file-name "backups/" dreams/cache-dir))))
  (setq auto-save-file-name-transforms `((".*" ,(expand-file-name "auto-save/" dreams/cache-dir) t)))
  (setq bookmark-default-file (expand-file-name "bookmarks" dreams/state-dir))
  (setq project-list-file (expand-file-name "projects" dreams/state-dir))
  (setq recentf-save-file (expand-file-name "recentf" dreams/state-dir))
  (setq savehist-file (expand-file-name "savehist" dreams/state-dir))
  (setq save-place-file (expand-file-name "places" dreams/state-dir))
  (setq url-history-file (expand-file-name "url-history" dreams/state-dir))
  (setq package-user-dir (expand-file-name "elpa" dreams/state-dir))
  (setq create-lockfiles nil)
#+end_src

* Projects and Dired

Use Emacs' built-in project manager as the project entry point and Dired as the
project landing page. =SPC p p= switches projects, remembers the selected root,
and opens Dired there.

#+begin_src emacs-lisp
  (require 'project)
  (require 'dired)
  (require 'dired-x)

  (setq project-switch-commands 'project-dired)

  (when (boundp 'project-vc-extra-root-markers)
    (setq project-vc-extra-root-markers '(".project")))

  (setq dired-dwim-target t)
  (setq dired-kill-when-opening-new-dired-buffer t)
  (setq dired-listing-switches "-alh --group-directories-first")

  (defun dreams/project-root ()
    "Return the current project root as a directory name."
    (if-let* ((project (project-current)))
        (project-root project)
      default-directory))

  (defun dreams/project-open-root ()
    "Open Dired at the current project root."
    (interactive)
    (dired (dreams/project-root)))

  (defun dreams/project-find-file ()
    "Fuzzy-find a file in the current project, or fall back to `find-file'."
    (interactive)
    (if (and (project-current) (fboundp 'consult-fd) (executable-find "fd"))
        (consult-fd (dreams/project-root))
      (if (project-current)
        (call-interactively #'project-find-file)
        (call-interactively #'find-file))))

  (defun dreams/project-search ()
    "Search the current project with Consult when available."
    (interactive)
    (if (and (project-current) (fboundp 'consult-ripgrep))
        (consult-ripgrep (dreams/project-root))
      (call-interactively #'project-find-regexp)))

  (defun dreams/project-switch-buffer ()
    "Switch to a project buffer with Consult when available."
    (interactive)
    (if (and (project-current) (fboundp 'consult-project-buffer))
        (call-interactively #'consult-project-buffer)
      (call-interactively #'project-switch-to-buffer)))

  (defun dreams/switch-buffer ()
    "Switch buffers with Consult when available."
    (interactive)
    (if (fboundp 'consult-buffer)
        (call-interactively #'consult-buffer)
      (call-interactively #'switch-to-buffer)))

  (defun dreams/recent-file ()
    "Open a recent file with Consult when available."
    (interactive)
    (if (fboundp 'consult-recent-file)
        (call-interactively #'consult-recent-file)
      (call-interactively #'recentf-open-files)))

  (defun dreams/project-eshell ()
    "Open Eshell at the current project root."
    (interactive)
    (let ((default-directory (dreams/project-root)))
      (eshell t)))
#+end_src

* Rust

Use tree-sitter for Rust when the grammar is available, otherwise fall back to
=rust-mode=. The Nix Emacs package closure provides the Rust and TOML grammars,
while the system language module already provides =rust-analyzer=. Rust Analyzer
knows how to provide dense inline hints, but they stay off by default because
they add a lot of visual noise; toggle them with =SPC l i= when needed.

#+begin_src emacs-lisp
  (setq treesit-font-lock-level 4)

  (setq-default eglot-workspace-configuration
                '(:rust-analyzer
                  (:cargo
                   (:allFeatures t)
                   :check
                   (:command "clippy")
                   :procMacro
                   (:enable t)
                   :inlayHints
                   (:bindingModeHints
                    (:enable t)
                    :chainingHints
                    (:enable t)
                    :closingBraceHints
                    (:enable t :minLines 10)
                    :closureReturnTypeHints
                    (:enable "always")
                    :discriminantHints
                    (:enable "always")
                    :expressionAdjustmentHints
                    (:enable "always" :hideOutsideUnsafe t)
                    :lifetimeElisionHints
                    (:enable "skip_trivial" :useParameterNames t)
                    :parameterHints
                    (:enable t)
                    :rangeExclusiveHints
                    (:enable t)
                    :renderColons t
                    :typeHints
                    (:enable t :hideClosureInitialization nil :hideNamedConstructor nil)))))

  (defun dreams/rust-mode ()
    "Use `rust-ts-mode' when tree-sitter Rust is ready, otherwise use `rust-mode'."
    (interactive)
    (if (and (fboundp 'rust-ts-mode)
             (require 'treesit nil t)
             (treesit-ready-p 'rust t))
        (rust-ts-mode)

          (rust-mode)
